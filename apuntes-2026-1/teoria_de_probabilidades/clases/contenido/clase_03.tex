\section{Integración y Esperanza}

Sean $(\Omega, \mcal{F}, \mu)$ un espacio de medida y $f \colon \Omega \to \R$ $(\mcal{F}, \beta(\R)$-medible. Entonces, definimos $\int_{\Omega} f \ d\mu$, la \textit{integral de $f$ respecto a $\mu$}, como:

\begin{enumerate}[i)]
	\item Si $f$ es \textit{simple}, i.e,
	\[ f = \sum_{i=1}^{k} a_{k} \mathds{1}_{A_{k}} \]
	con $(A_{i})_{i = 1,\dots,k} \subseteq \mcal{F}$ partición finita de $\Omega$, entonces
	\[ \int_{\Omega} f \ d\mu \coloneq \sum_{i=1}^{k} a_{k} \mu(A_{k}). \]

	\item Si $f \geq 0$,
	\[ \int_{\Omega} f \ d\mu \coloneq \sup \Big\{ \int_{\Omega} g \ d\mu \ : \ g \text{ simple, } 0 \leq g \leq f \Big\} \in [0,\infty]. \]

	\item Si $f$ cambia de signo, consideramos
	\[ \begin{cases}
		f^{+}(\omega) \coloneq f(\omega) \mathds{1}_{f(\omega) \geq 0} \\
		f^{-}(\omega) \coloneq -f(\omega) \mathds{1}_{f(\omega) < 0}
	\end{cases} \]
	y, si $\int_{\Omega} f^{+} \ d\mu < \infty$ ó $\int_{\Omega} f^{-} \ d\mu < \infty$, definimos
	\[ \int_{\Omega} f \ f\mu \coloneq \int_{\Omega} f^{+} \ d\mu - \int_{\Omega} f^{-} \ d\mu \in [-\infty,\infty]. \]

	\item Si $\int_{\Omega} f^{+} \ d\mu = \int_{\Omega} f^{-} d\mu = \infty$, la integral de $f$ respecto de $\mu$ no se encuentra bien definida.
\end{enumerate}

\noindent Además,

\begin{itemize}
	\item Diremos que $f$ es \textit{$\mu$-integrable} (o absolutamente integrable) si
	\[ \int_{\Omega} |f| \ d\mu < +\infty \]
	($\iff \int_{\Omega} f^{+} \ d\mu < \infty$ y $\int_{\Omega} f^{-} \ d\mu < \infty$) y \textit{débilmente $\mu$-integrable} si
	\[ \min \Big\{ \int_{\Omega} f^{+} \ d\mu, \ \int_{\Omega} f^{-} \ d\mu \Big\} < \infty. \]

	\item Por último, si $(\Omega, \mcal{F}, \mbb{P})$ es un espacio de probabilidad y $X$ es una variable aleatoria (a valores en $\R$) en este espacio, definimos $\mbb{E}(X)$, la \textit{esperanza de $X$} (o \textit{media de $X$}), como
	\[ \mbb{E}(X) \coloneq \int_{\Omega} X \ d\mbb{P} \]
	siempre que la integral se encuentre bien definida.
\end{itemize}

\begin{note}
	Valen para la esperanza todas las propiedades de la integral que ya se conocen (linealidad, monotonía, teoremas de paso al límite, etc.)
\end{note}

\begin{definition}
	Sean $\mu$ y $\nu$ dos medidas sobre $(\Omega, \mcal{F})$. Diremos que:
	\begin{enumerate}
		\item $\nu$ es \textit{absolutamente continua} respecto de $\mu$, y lo notamos $\nu \ll \mu$, si $\mu(A) = 0 \implies \nu(A) = 0$.

		\item $\mu$ y $\nu$ son \textit{equivalentes} si $\nu \ll \mu$ y $\mu \ll \nu$.

		\item $\nu$ es \textit{singular} respecto de $\mu$, y lo notamos $\nu \perp \mu$, si existe $A \in \mcal{F}$ tal que $\mu(A) = 0$ y $\nu(A^{c}) = 0$.
	\end{enumerate}
\end{definition}

\begin{eg}~
	\begin{enumerate}
		\item $X$ discreta $\implies \mbb{P}_{X} \perp$ Lebesgue (no vale la vuelta!);

		\item $X$ absollutamente continua $\iff \mbb{P}_{X} \ll$ Lebesgue;

		\item $X$ discreta $\iff \mbb{P}_{X} \ll N_{\mcal{A}_{X}}$ (medida de contar en $\mcal{A}_{X}$).
	\end{enumerate}
\end{eg}

\begin{theorem}[Radon-Nikodym]
    Sean \(\mu\) y \(\nu\) dos medidas \(\sigma\)-finitas sobre \((E, \Sigma)\). Entonces las siguientes afirmaciones son equivalentes:
    \begin{enumerate}
        \item \(\nu \ll \mu\)

		\item Existe una función \(f: E \to \mathbb{R}_{\ge 0}\) \((\mcal{F}, \beta(\mathbb{R}))\)-medible tal que
        \[
		\nu(A) = \int_{A} f(\omega) \, d\mu(\omega) \quad \text{para todo } A \in \mcal{F}.
        \]
    \end{enumerate}
    La función \(f\) es única salvo conjuntos de \(\mu\)-medida nula y se la llama \textit{derivada de Radon-Nikodym} de \(\nu\) respecto de \(\mu\), denotada por \(f = \frac{d\nu}{d\mu}\).
\end{theorem}

\section{Teorema de existencia y unicidad de variable aleatoria con distribución dada}

Volvamos al teorema de la clase pasada para demostrarlo:

\begin{theorem}
    Sea \(F\) una función de distribución. Entonces:
    \begin{enumerate}
        \item Existe una variable aleatoria \(X\) (definida en algún espacio de probabilidad) tal que \(F = F_X\).
        \item La distribución de dicha variable es única, i.e.
        \[
		F_{X_1} = F_{X_2} \implies \mathbb{P}_{X_1} = \mathbb{P}_{X_2} \]
		$(\text{si } \mathbb{P}_{X_1} = \mathbb{P}_{X_2} \text{ lo notamos } X_1 \overset{\mathcal{L}}{=} X_2)$.
    \end{enumerate}
\end{theorem}

\begin{corollary}~
    \begin{enumerate}
        \item Dada una probabilidad \(\mathbb{P}\) sobre \((\mathbb{R}, \beta(\mathbb{R}))\), la función \(F: \mathbb{R} \to [0,1]\) definida por \(F(x) := \mathbb{P}((-\infty, x])\) es una función de distribución.

        \item Dada una función de distribución \(F\), existe una única probabilidad \(\mathbb{P}\) sobre \((\mathbb{R}, \beta(\mathbb{R}))\) tal que \(\mathbb{P}((-\infty, x]) = F(x)\) para todo \(x \in \mathbb{R}\). De hecho, \(\mathbb{P} = \mathbb{P}_X\) donde \(X\) es la variable del teorema.
    \end{enumerate}
\end{corollary}
\medskip \par
Demostraremos únicamente el teorema; la deducción del corolario a partir de éste es sencilla. Además, lo haremos de dos formas:

\begin{enumerate}
    \item \textbf{Forma canónica}: tomamos \((\Omega, \mathcal{F}) := (\mathbb{R}, \beta(\mathbb{R}))\) y \(X := \operatorname{id}_\Omega\) (\(X(\omega) = \omega\)) y definimos \(\mathbb{P}\) sobre \((\Omega, \mathcal{F})\) de manera tal que en \((\Omega, \mathcal{F}, \mathbb{P})\) la variable \(X := \operatorname{id}_\Omega\) cumpla lo buscado.
	\item \textbf{Forma constructiva}: tomamos \((\Omega, \mathcal{F}, \mathbb{P}) := ((0,1), \beta((0,1)), \operatorname{Leb}_{(0,1)})\) y definimos allí una variable \(X\) que cumpla lo buscado.
\end{enumerate}

\subsection{Forma canónica:}

Para esta forma, utilizamos el Teorema de Carathéodory.
\begin{definition}
	Una clase \(\mscr{S}\) de subconjuntos de un espacio \(E\) se dice una \textit{semiálgebra} si satisface las siguientes propiedades:
	\begin{enumerate}[i)]
		\item \(\mscr{S}\) es no vacía;

        \item \(S, T \in \mscr{S} \Rightarrow S \cap T \in \mscr{S}\) (es cerrada por intersecciones);

		\item \(S \in \mscr{S} \Rightarrow \exists T_1, \dots, T_n \in \mscr{S}\) disjuntos tales que \(S^{c} = \bigcup_{i=1}^n T_i\).
    \end{enumerate}
\end{definition}

\begin{remark}
	\(\varnothing \in \mscr{S}\): si \(S \in \mscr{S}\) y \(S = \bigcup_{i=1}^n T_i\) con \(T_i\) disjuntos, \(\varnothing = S \cap \varnothing \in \mscr{S}\) por (ii).
\end{remark}

\begin{theorem}[Extensión de Carathéodory]
	Sean \(\mscr{S}\) una semialgebra de subconjuntos de un espacio \(E\) y \(\mu_0: \mscr{S} \to [0,1]\) una aplicación que satisface:
    \begin{enumerate}
		\item[S0)] \(\mu_0(\varnothing) = 0\).

		\item[S1)] \(\mu_0\) es finitamente aditiva en \(\mscr{S}\):
        \[
        (A_n)_{n=1,\dots,N} \subseteq \mscr{S} \text{ disjuntos, } \bigcup_{n=1}^N A_n \in \mscr{S} \implies \mu_0\!\left(\bigcup_{n=1}^N A_n\right) = \sum_{n=1}^N \mu_0(A_n).
        \]

		\item[S2)] \(\mu_0\) es \(\sigma\)-subaditiva en \(\mscr{S}\):
        \[
        (A_n)_{n\in\mathbb{N}} \subseteq \mscr{S} \text{ disjuntos, } A \in \mscr{S}, A \subseteq \bigcup_{n\in\mathbb{N}} A_n \implies \mu_0(A) \leq \sum_{n\in\mathbb{N}} \mu_0(A_n).
        \]
    \end{enumerate}
    Entonces existe una medida \(\mu\) sobre \((E, \sigma(\mscr{S}))\) tal que \(\mu|_{\mscr{S}} = \mu_0\).
    \par
    Además, si \(\mu_0\) es \(\sigma\)-finita, entonces la extensión \(\mu\) es única; donde $\sigma$-finita significa que existen \((E_n)_{n\in\mathbb{N}} \subseteq \mscr{S}\) tales que \(E = \bigcup_n E_n\) y \(\mu_0(E_n) < \infty\) para todo \(n\).
\end{theorem}

\subsubsection{Idea de demostración del teorema (forma canónica)}

\begin{itemize}
	\item Consideramos la semialgebra \(\mscr{S} = \{(a,b] : -\infty \le a \le b \le \infty\}\) con la convención \((a,a] = \varnothing\) y \((a,\infty)\) si \(b=\infty\).

    \item Definimos \(\mu_0: \mscr{S} \to [0,1]\) como \(\mu_0((a,b]) = F(b) - F(a)\).

    \item Probamos que \(\mu_0\) satisface (s0), (s1), (s2). (La demostración de (s2) requiere trabajo!)

    \item Por el teorema de Carathéodory, existe extensión \(\mu\) a todo \(\sigma(\mscr{S}) = \beta(\mathbb{R})\).

    \item Tomamos \(\mathbb{P} := \mu\) y vemos que en el espacio de probabilidad \((\mathbb{R}, \beta(\mathbb{R}), \mathbb{P})\) la variable aleatoria \(X := \operatorname{id}_{\mathbb{R}}\) cumple que
    \[
    F_X(x) = \mathbb{P}(\{\omega : X(\omega) \le x\}) = \mathbb{P}((-\infty, x]) = F(x) - F(-\infty) = F(x).
    \]

	\item Si \(F_{X_{1}} = F_{X_{2}} = F\), entonces \(\mathbb{P}_{X_{1}}\) y \(\mathbb{P}_{X_{2}}\) son extensiones de \(\mu_0\) a \(\sigma(\mscr{S})\). Como \(\mu_0(\mathbb{R}) = F(\infty) - F(-\infty) = 1\), por la unicidad en el teorema de Carathéodory, \(\mathbb{P}_{X_{1}} = \mathbb{P}_{X_{2}}\).
\end{itemize}

\begin{remark}
    La forma canónica lleva bastante trabajo (que no hicimos), que es el mismo que se suele hacer al construir la medida de Lebesgue. La forma constructiva es más rápida (y útil), pero asume que ya tenemos construida la medida de Lebesgue de antemano.
\end{remark}

\subsection{Forma constructiva: inversa generalizada}

Para la demostración de forma constructiva, introducimos la inversa generalizada.

\begin{definition}[Inversa generalizada]
    Sea \(F: \mathbb{R} \to [0,1]\) una función. Definimos su \textit{inversa generalizada} \(F^{-1}: [0,1] \to \overline{\mathbb{R}}\) mediante la fórmula
    \[
    F^{-1}(y) := \sup \{z \in \mathbb{R} : F(z) < y\},
    \]
    donde convenimos que \(\sup \varnothing = -\infty\) y \(\sup A = \infty\) si \(A\) no está acotado superiormente.
\end{definition}

\begin{lemma}
	Sea \(F: \mathbb{R} \to [0,1]\) una función de distribución. Entonces, si \(U\) es una variable aleatoria tal que \(\mathbb{P}_U = \mcal{L}|_{[0,1]}\), i.e. \(\mathbb{P}(U \in B) = |B \cap [0,1]|\) para todo Borel \(B\), resulta que \(X := F^{-1}(U)\) es una variable aleatoria y satisface \(F_X = F\).
\end{lemma}

\begin{proof}
    \begin{enumerate}
        \item En primer lugar, observemos que \(F^{-1}\) es medible Borel por ser monótona creciente por definición.

        \item En particular, \(X := F^{-1}(U)\) es variable aleatoria (pues \(X^{-1}(B) = U^{-1}((F^{-1})^{-1}(B)) \in \mathcal{F}\) para todo \(B \in \beta(\mathbb{R})\)).

        \item Además, vale que
        \[
        \{X \le x\} = \{U \le F(x)\} \quad \forall x \in \mathbb{R}, \tag{1}
        \]
        de manera tal que
		\begin{align*}
			F_X(x) = \mathbb{P}(X \le x) &= \mathbb{P}(U \le F(x)) \\
			&= |(-\infty, F(x)] \cap [0,1]| \\
			&= |[0, F(x)]| \\
			&= F(x).
		\end{align*}
        En efecto, para ver (1) notemos:
        \begin{itemize}
            \item[($\subseteq$)] Si \(\omega \in \Omega\) es tal que \(X(\omega) \le x\) entonces debe valer que \(U(\omega) \le F(x)\). En caso contrario, si \(F(x) < U(\omega)\), como \(F\) es continua a derecha, existiría \(\varepsilon > 0\) tal que \(F(x+\varepsilon) < U(\omega)\). En particular, \(x+\varepsilon \in \{z \in \mathbb{R} : F(z) < U(\omega)\}\) y, por lo tanto, 
			\[ x+\varepsilon \le \sup\{z : F(z) < U(\omega)\} = F^{-1}(U(\omega)) = X(\omega), \]
			lo que implica \(X(\omega) > x\), y esto contradice nuestra suposición..

            \item[($\supseteq$)] Si \(\omega\) cumple que \(U(\omega) \le F(x)\), entonces 
			\[ x \not\in \{z \in \mathbb{R} : F(z) < U(\omega)\} \eqcolon I_{\omega}. \]
			En efecto, si \(I_{\omega} \neq \varnothing\), entonces $I_{\omega}$ es un intervalo no acotado inferiormente por ser monótona creciente (i.e. si $z \in I_{\omega}$ y $z' \leq z$, entonces $F(z') \leq F(z) < U(\omega)$ y, por ende, $z' \in I_{\omega}$), entonces $x$ es cota superior de $I_{\omega}$ y, por lo tanto, 
			\[ X = F^{-1}(U(\omega)) = \sup I_{\omega} \leq x \]
			y así vemos que vale la contensión.
        \end{itemize}
        Así queda demostrada la igualdad (1).
    \end{enumerate}
\end{proof}

\begin{remark}
	\(F^{-1}\) se llama inversa generalizada de \(F\) puesto que satisface la igualdad \(\{y : F^{-1}(y) \le x\} = \{y : y \le F(x)\}\) dada en (1) y, si en vez de \(\le\) tuviésemos \(=\) en (1), entonces sería la inversa tradicional.
\end{remark}

Ahora estamos listos para dar la demostración de forma constructiva.

\begin{proof}[Proof][Demostración del Teorema-(1), Forma constructiva]
	Tomamos \((\Omega, \mathcal{F}, \mathbb{P}) := ([0,1], \beta([0,1]), \mcal{L}|_{[0,1]})\) y \(U := \operatorname{id}_{[0,1]}\). Como \(\mathbb{P}_U = \mcal{L}|_{[0,1]}\) por definición, por el lema previo, \(X := F^{-1}(U)\) es la variable aleatoria buscada.
\end{proof}
\medskip \par
Para demostrar la unicidad (teorema, parte 2) necesitamos otro resultado.

\begin{definition}[\(\pi\)-sistema]
    Una clase \(\mathcal{P}\) de subconjuntos de un espacio \(E\) se dice un \textit{\(\pi\)-sistema} si es cerrada por intersecciones finitas, i.e., si \(A, B \in \mathcal{P}\) entonces \(A \cap B \in \mathcal{P}\).
\end{definition}

\begin{definition}[\(\lambda\)-sistema]
    Una clase \(\mathcal{D}\) de subconjuntos de un espacio \(E\) se dice un \textit{\(\lambda\)-sistema} si verifica las siguientes condiciones:
    \begin{enumerate}[I)]
        \item \(E \in \mathcal{D}\).
        \item \(A \in \mathcal{D} \Rightarrow A^c \in \mathcal{D}\).
        \item \((A_n)_{n\in\mathbb{N}} \subseteq \mathcal{D}\) disjuntos \(\Rightarrow \bigcup_{n\in\mathbb{N}} A_n \in \mathcal{D}\).
    \end{enumerate}
\end{definition}

\begin{theorem}[\(\pi\)-\(\lambda\) de Dynkin]
    Si \(\mathcal{P}\) es un \(\pi\)-sistema y \(\mathcal{D}\) es un \(\lambda\)-sistema tales que \(\mathcal{P} \subseteq \mathcal{D}\), entonces \(\sigma(\mathcal{P}) \subseteq \mathcal{D}\).
\end{theorem}

\begin{proof}
	Ver Apéndice I o [Teorema 3.2] de Billingsley (P \& M).
\end{proof}
\medskip \par
Como consecuencia, tenemos el siguiente corolario.

\begin{corollary}
    Supongamos que \(\mathbb{P}_1\) y \(\mathbb{P}_2\) son probabilidades en \(\sigma(\mathcal{P})\), donde \(\mathcal{P}\) es un \(\pi\)-sistema. Si \(\mathbb{P}_1\) y \(\mathbb{P}_2\) coinciden en \(\mathcal{P}\) entonces también coinciden en todo \(\sigma(\mathcal{P})\).
\end{corollary}

\begin{proof}
    Sea \(\mathcal{D} = \{ A \in \sigma(\mathcal{P}) : \mathbb{P}_1(A) = \mathbb{P}_2(A) \}\).
    \begin{itemize}
        \item Por hipótesis, \(\mathcal{P} \subseteq \mathcal{D}\).
        \item Si vemos que \(\mathcal{D}\) es un \(\lambda\)-sistema, entonces por el teorema de Dynkin tendremos que \(\sigma(\mathcal{P}) \subseteq \mathcal{D}\) y, por lo tanto, \(\mathbb{P}_1 \equiv \mathbb{P}_2\) en \(\sigma(\mathcal{P})\).
    \end{itemize}
    Veamos que \(\mathcal{D}\) es un \(\lambda\)-sistema:
	\begin{enumerate}[i)]
        \item \(\mathbb{P}_1(\Omega) = 1 = \mathbb{P}_2(\Omega)\) por ser probabilidades. Luego \(\Omega \in \mathcal{D}\).
        \item \(A \in \mathcal{D} \Rightarrow \mathbb{P}_1(A) = \mathbb{P}_2(A) \Rightarrow 1 - \mathbb{P}_1(A) = 1 - \mathbb{P}_2(A) \Rightarrow \mathbb{P}_1(A^c) = \mathbb{P}_2(A^c) \Rightarrow A^c \in \mathcal{D}\).
        \item Si \((A_n)_{n\in\mathbb{N}} \subseteq \mathcal{D}\) son disjuntos, entonces
        \[
        \mathbb{P}_1\!\left(\bigcup_{n} A_n\right) = \sum_{n} \mathbb{P}_1(A_n) = \sum_{n} \mathbb{P}_2(A_n) = \mathbb{P}_2\!\left(\bigcup_{n} A_n\right),
        \]
        por la \(\sigma\)-aditividad de las probabilidades. Luego \(\bigcup_n A_n \in \mathcal{D}\).
    \end{enumerate}
    Luego \(\mathcal{D}\) es un \(\lambda\)-sistema y esto concluye la demostración.
\end{proof}
\medskip \par
Ahora estamos listos para terminar la demostración del Teorema.

\begin{proof}[Proof ][Demostración de Teorema-(2), Forma Constructiva]
	Dado $x \in \R$ y dos variables aleatorias $X_{1}$ y $X_{2}$ (no necesariamente definidas en el mismo EdP)
	\begin{align*}
		F_{X_{1}}(x) = F_{X_{2}}(x) &\iff \mbb{P}(X_{1} \leq x) = \mbb{P}(X_{2} \leq x) \\
		&\iff \mbb{P}(X_{1} \in (-\infty,x]) = \mbb{P}(X_{2} \in (-\infty,x]) \\
		&\iff \mbb{P}_{X_{1}}((-\infty,x]) = \mbb{P}_{X_{2}}((-\infty,x])
	.\end{align*}
	Como $\mcal{P} = \{(-\infty,x] \ : \ x \in \R\}$ es un $\pi$-sistema, por el corolario anterior, tenemos que $F_{X_{1}} \equiv F_{X_{2}} \implies \mbb{P}_{X_{1}} = \mbb{P}_{X_{2}}$ en $\sigma(\mcal{P})$. Luego, para concluir sólo resta ver que $\sigma(\mcal{P}) = \beta(\R)$. \par
	$\boxed{\subseteq}$ $\beta(\R)$ contiene a todos los abiertos y es cerrada por complementos. Luego, $\beta(\R)$ contiene a todos los cerrados y, en particular, a todos los elementos de $\mcal{P}$. Por minimalidad de $\sigma(\mcal{P})$, $\sigma(\mcal{P}) \subseteq \beta(\R)$. \par
	$\boxed{\supseteq}$ $\sigma(\mcal{P})$ contiene a todos los intervalos de la forma $(a,b] = (-\infty, b] \cap ((-\infty, a])^{c}$ y, por lo tanto, a todos los intervalos abiertos $(a,b) = \bigcup_{n \in \N} (a,b - \frac{1}{n}]$. Como cualquier abierto en $\R$ es unión numerable de intervalos $(a,b)$ (por ser $\R$ separable), $\sigma(\mcal{P})$ contiene a cualquier abierto en $\R$. Por minimalidad de $\beta(\R)$, resulta que $\beta(\R) \subseteq \sigma(\mcal{P})$.
\end{proof}
