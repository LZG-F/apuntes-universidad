\section{Ley 0-1 de Kolmogorov}

Dada una sucesión \((X_n)_{n\in\mathbb{N}}\) de variables aleatorias independientes, definimos la \(\sigma\)-álgebra \textit{cola} asociada a \((X_n)_{n\in\mathbb{N}}\) como
\[
\mscr{T} = \bigcap_{n\in\mathbb{N}} \sigma(X_i : i \ge n).
\]
Los eventos en \(\mscr{T}\) se llaman \textit{eventos cola}.

La idea es que un evento cola queda determinado únicamente por el comportamiento de la cola de la sucesión \((X_n)_{n\in\mathbb{N}}\).

\begin{eg}
    Los siguientes son eventos cola:
    \begin{enumerate}[i)]
        \item \(A = \{\displaystyle \lim_{n\to\infty} X_n = 5\}\).
        \item \(B = \{(X_n)_{n\in\mathbb{N}} \text{ es de Cauchy}\}\).
        \item \(C = \{(X_n)_{n\in\mathbb{N}} \text{ es acotada}\}\) (si cada \(X_n\) toma valores reales y no pueden valer \(\pm\infty\)).
    \end{enumerate}
\end{eg}

\noindent Para eventos cola de \((X_n)_{n\in\mathbb{N}}\) independientes, tenemos:

\begin{theorem}[Ley 0-1 de Kolmogorov]
	Si \((X_n)_{n\in\mathbb{N}}\) son independientes y \(A \in \mscr{T}\) es un evento cola, entonces \(\mathbb{P}(A) = 0\) ó \(\mathbb{P}(A) = 1\).
\end{theorem}

\begin{proof}[Demostración (esquema)]
    Sean \(\mathcal{F}_n = \sigma(X_{i} \ : \ 1 \leq i \leq n)\) y \(\mathcal{G}_n = \sigma(X_{i} \ : \ i > n)\). Observar que si para \(I \subseteq \mathbb{N}\) definimos
    \[
    \mathcal{P}_I = \Big\{ \bigcap_{j \in J} \{X_j \in B_j\} \ : \ B_j \in \beta(\mathbb{R}),\; J \subseteq I \text{ finito} \Big\},
    \]
    entonces:
    \begin{itemize}
        \item \(\mathcal{P}_I\) es un \(\pi\)-sistema.

        \item \(\sigma(\mathcal{P}_I) = \sigma(X_i : i \in I)\).

        \item si \(I \cap I' = \varnothing\) entonces \(\mathcal{P}_I\) y \(\mathcal{P}_{I'}\) son independientes.
	\end{itemize}
	Se sigue del teorema visto la clase pasada que si \(I \cap I' = \varnothing\), entonces \(\sigma(\mathcal{P}_I)\) y \(\sigma(\mathcal{P}_{I'})\) también lo son. \par
	En particular, \(\mathcal{F}_n = \sigma(\mcal{P}_{\{1,\dots,n\}})\) y \(\mathcal{G}_n = \sigma(\mcal{P}_{\{n, n+1, \dots\}})\) son independientes para todo \(n\). Luego, \(\mathcal{G}_{\infty}\) es independiente de $\mcal{F}_{n}$ para todo $n \in \N$. \par
	En particular, $\mcal{G}_{\infty}$ es independiente de $\bigcup_{n \in \N} \mcal{F}_{n}$ y, como $\bigcup_{n \in \N} \mcal{F}_{n}$ es un $\pi$-sistema, por el mismo Teorema $\mcal{G}_{\infty}$ es independiente de $\sigma(\bigcup_{n \in \N} \mcal{F}_{n}) = \mcal{F}_{\infty}$. \par
	Luego, si $A \in \mcal{G}_{\infty}$, entonces 
	\[ \mbb{P}(A) = \mbb{P}(\underbrace{A}_{\in \mcal{G}_{\infty}} \cap \underbrace{A}_{\in \mcal{F}_{\infty}}) = \mbb{P}(A) \mbb{P}(A) \]
	y, por ende, $\mbb{P}(A) \in \{0,1\}$.
\end{proof}

\section{Lemas de Borel-Cantelli}

\begin{definition}
    Dada una sucesión \((A_n)_{n\in\mathbb{N}} \subseteq \mathcal{F}\) de eventos, definimos el evento
	\begin{align*}
		\limsup_{n\to\infty} A_n &\coloneq \{\text{ocurren infinitos } A_n\} \\
		&= \{\omega \in \Omega \ : \ \omega \in A_{n} \text{ para infinitos } n\} \\
		&= \bigcap_{N \in \N} \bigcup_{n = N}^{\infty} A_{n}.
	\end{align*}
    Notar que \(\limsup A_n \in \mathcal{F}\). De hecho, se tiene que \(\mathds{1}_{\limsup A_n} = \limsup_{n\to\infty} \mathds{1}_{A_n}\) (y por este motivo se llama \(\limsup A_{n}\)).
\end{definition}

\begin{lemma}[Borel-Cantelli]
    Sea \((A_n)_{n\in\mathbb{N}}\) una sucesión de eventos. Entonces:
    \begin{itemize}
		\item[I)] Si \(\sum_{n\in\mathbb{N}} \mathbb{P}(A_n) < \infty\) entonces 
		\[ \mathbb{P}(\text{ocurren infinitos } A_n) = 0. \]

		\item[II)] Si \(\sum_{n\in\mathbb{N}} \mathbb{P}(A_n) = \infty\) y \((A_n)_{n\in\mathbb{N}}\) son independientes, entonces 
		\[ \mathbb{P}(\text{ocurren infinitos } A_n) = 1. \]
    \end{itemize}
\end{lemma}

\begin{remark}
	El evento $\{\text{Ocurren infinitos } A_{n}\}$ es un evento cola para $(\mathds{1}_{A_{n}})_{n \in \N}$. En efecto, esto se sigue del hecho que $\mathds{1}_{\{\text{Ocurren } \infty's \ A_{n}\}} = \limsup_{n \to \infty} \mathds{1}_{A_{n}}$. Luego, por la Ley 0-1 de Kolmogorov, si los $(A_{n})_{n \in \N}$ son independientes, tenemos que $\mbb{P}(\text{Ocurren } \infty's \ A_{n}) \in \{0,1\}$. \par
	El lema de Borel Cantelli caracteriza cuando vale 0 ó 1 en términos de $\sum_{n \in \N} \mbb{P}(A_{n})$.
\end{remark}

\begin{note}
	La hipótesis  de independencia en BC-II es \textbf{necesaria}. En efecto, si tomamos $A \in \mcal{F}$ como $\mbb{P}(A) \in (0,1)$ y definimos $A_{n} \coloneq A \quad \forall n$, entonces 
	\[ \limsup_{n \to \infty} A_{n} = A \]
	y, por lo tanto,
	\[ \mbb{P}(\text{Ocurren } \infty's \ A_{n}) = \mbb{P}(A) \in (0,1). \]
\end{note}

\begin{proof}[Proof Other Information][Lemas de Borel-Cantelli]
	$\boxed{I.}$ Si definimos $N \coloneq \sum_{n \in \N} \mathds{1}_{A_{n}}$ entonces, por el Teorema de Convergencia Monótona,
	\[ \mbb{E}(N) = \sum_{n \in \N} \mbb{P}(A_{n}) < \infty. \]
	En particular,
	\[ \mbb{P}(\text{Ocurren } \infty's \ A_{n}) = \mbb{P}(N = \infty) = 0. \]
	\par $\boxed{II.}$ Basta con probar que $\mbb{P}(\text{Ocurren finitos } A_{n}) = 0$. Pero,
	\[ \{\text{Ocurren finitos } A_{n}\} = (\limsup_{n \to \infty} A_{n})^{c} = \bigcup_{N \in \N} \bigcap_{m \geq N} A_{m}^{c}, \]
	con lo cual bastará con ver que $\mbb{P}(\bigcap_{m \geq N} A_{m}^{c}) = 0$ para todo $N \in \N$. Por continuidad superior de $\mbb{P}$,
	\[ \mbb{P} \Big( \bigcap_{m \geq N} A_{m}^{c} \Big) = \lim_{k \to \infty} \mbb{P} \Big( \bigcap_{m = N}^{k} A_{m}^{c} \Big). \]
	Pero,
	\begin{align*}
		\mbb{P} \Big( \bigcap_{m = N}^{k} A_{m}^{c} \Big) &= \prod_{m = N}^{k} \mbb{P}(A_{m}^{c}) \\
		&= \prod_{m = N}^{k} 1 - \mbb{P}(A_{m}) \\
		&\leq e^{-\sum_{m=N}^{k} \mbb{P}(A_{m})} \xlongrightarrow[k \to \infty]{} 0
	,\end{align*}
	si $\sum_{m \in \N} \mbb{P}(A_{m}) = \infty$. (notar que la primera igualdad está dada pues los $A_{m}^{c}$ son independientes, pues los $\mathds{1}_{A_{n}}$ son independientes, y que la desigualdad está dada por $1 - x \leq e^{-x} \quad \forall x \in \R$.)
\end{proof}

\begin{remark}
	El argumento de arriba muestra, en particular, que si $(B_{n})_{n \in \N}$ so independientes, entonces
	\[ \mbb{P} \Big( \bigcap_{n \in \N} B_{n} \Big) = \prod_{n \in \N} \mbb{P}(B_{n}). \]
\end{remark}

\section{Existencia de sucesiones independientes}

¿Existe realmente una sucesión \((X_n)_{n\in\mathbb{N}}\) de variables aleatorias independientes? (o, equivalentemente, de eventos \((A_n)_{n\in\mathbb{N}}\) independientes?). \par
Es decir, ¿existe un espacio de probabilidad \((\Omega, \mathcal{F}, \mathbb{P})\) y una sucesión \((X_n)_{n\in\mathbb{N}}\) de variables aleatorias definidas sobre \((\Omega, \mathcal{F})\) tal que las \((X_n)_{n\in\mathbb{N}}\) sean independientes bajo \(\mathbb{P}\)?

\begin{eg}
    Basta considerar el caso en que las \(X_n\) son uniformes en \((0,1)\); a partir de allí podemos cubrir cualquier otro caso.
\end{eg}

\begin{itemize}
	\item ¿Cómo construimos una variable \(U \sim \text{Uniforme}(0,1)\)? Como \(\mathbb{P} = \mscr{L}|_{[0,1]}\), construir \(U\) es equivalente a construir la medida de Lebesgue en \([0,1]\).

	\item ¿Cómo conseguimos \(U_1,\dots,U_n\) independientes con distribución \(U_{i} \sim \mcal{U}([0,1])\)? Por el teorema anterior,
	\[
	\mathbb{P}_{U_1,\dots,U_n} = \mathbb{P}_{U_1} \times \cdots \times \mathbb{P}_{U_n} = \mscr{L}|_{[0,1]} \times \cdots \times \mscr{L}|_{[0,1]} = \mscr{L}|_{[0,1]^{n}}.
	\]
	Por lo tanto, construir \((U_1,\dots,U_n)\) es equivalente a construir $\mscr{L}|_{[0,1]^{n}}$. Siguiendo esta lógica, construir \((U_n)_{n\in\mathbb{N}}\) independientes con \(U_n \sim \mcal{U}([0,1])\) es equivalente a construir la medida de Lebesgue en el cubo unitario infinito-dimensional \([0,1]^{\mathbb{N}}\). Esto no es para nada trivial, pero puede hacerse. Lo haremos de dos formas:
    \begin{enumerate}
        \item \textbf{Forma canónica}: construiremos \(\operatorname{leb}_{[0,1]^n}\) con el teorema de Carathéodory. (No ahora.)
		\item \textbf{Forma constructiva}: construiremos \((U_n)_{n\in\mathbb{N}}\) en \(([0,1], \beta([0,1]), \mscr{L}|_{[0,1]})\). Ejercicio Práctica 2.
    \end{enumerate}
\end{itemize}
